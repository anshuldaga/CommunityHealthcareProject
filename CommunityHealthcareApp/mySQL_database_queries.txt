CREATE TABLE `healthresource` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `serviceName` varchar(45) DEFAULT NULL,
  `agencyName` varchar(45) DEFAULT NULL,
  `website` varchar(45) DEFAULT NULL,
  `contactNumber` varchar(11) DEFAULT NULL,
  `contactAddress` varchar(45) DEFAULT NULL,
  `city` varchar(15) DEFAULT NULL,
  `zip` varchar(6) DEFAULT NULL,
  `description` varchar(45) DEFAULT NULL,
  `dayOpen` varchar(45) DEFAULT NULL,
  `eligibility` varchar(10) DEFAULT NULL,
  `timeOpen` time DEFAULT NULL,
  `timeClose` time DEFAULT NULL,
  `application` varchar(15) DEFAULT NULL,
  `busRotes` varchar(45) DEFAULT NULL,
  `translationalAbility` varchar(45) DEFAULT NULL,
  `translationLanguage` varchar(45) DEFAULT NULL,
  `cost` decimal(10,0) DEFAULT NULL,
  `unitedWayApproval` bit(1) DEFAULT NULL,
  `serviceArea` varchar(15) DEFAULT NULL,
  `moreInfo` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci COMMENT='	';

CREATE TABLE `medication` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `medicationName` varchar(45) DEFAULT NULL,
  `medicationFreq` int(11) DEFAULT NULL,
  `medicationDay` varchar(10) DEFAULT NULL,
  `enteredDate` datetime DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `usercontact` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `userId` int(11) NOT NULL,
  `userContactType` varchar(10) DEFAULT NULL,
  `userContactValue` varchar(45) DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `userId_idx` (`userId`),
  CONSTRAINT `id` FOREIGN KEY (`userId`) REFERENCES `usercredentials` (`id`) ON DELETE CASCADE ON UPDATE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `usercredentials` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `username` varchar(30) DEFAULT NULL,
  `password` varchar(30) DEFAULT NULL,
  PRIMARY KEY (`id`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `userdetail` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `userId` int(11) NOT NULL,
  `firstName` varchar(20) NOT NULL,
  `middleName` varchar(20) DEFAULT NULL,
  `lastName` varchar(20) DEFAULT NULL,
  `dateOfBirth` date NOT NULL,
  `SSN` varchar(12) DEFAULT NULL,
  `street` varchar(45) DEFAULT NULL,
  `houseNumber` varchar(10) DEFAULT NULL,
  `city` varchar(30) DEFAULT NULL,
  `zipcode` varchar(5) DEFAULT NULL,
  `state` varchar(2) DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `id_idx` (`userId`),
  CONSTRAINT `userId` FOREIGN KEY (`userId`) REFERENCES `usercredentials` (`id`) ON DELETE CASCADE ON UPDATE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

CREATE TABLE `userhealth` (
  `id` int(11) NOT NULL AUTO_INCREMENT,
  `userId` int(11) NOT NULL,
  `dateOfEntry` date DEFAULT NULL,
  `healthParameter` varchar(10) DEFAULT NULL,
  `healthValue` varchar(10) DEFAULT NULL,
  PRIMARY KEY (`id`),
  KEY `UserId_idx` (`userId`),
  CONSTRAINT `uId` FOREIGN KEY (`userId`) REFERENCES `usercredentials` (`id`) ON DELETE CASCADE ON UPDATE RESTRICT
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
